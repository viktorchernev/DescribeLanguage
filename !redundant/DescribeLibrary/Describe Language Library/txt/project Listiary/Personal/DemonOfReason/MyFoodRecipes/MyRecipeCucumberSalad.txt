cucumber salad <MyRecipeCucumberSalad> ->

	ingredients ->									//per 100g	//in this recipe

		1 cucumber,									//15		30
		2 table spoons of sunflower oil,			//884		120
		some salt;

	preparation ->

		cut the cucumber to half at its longitude,
		cut it to small pieces,
		add salt and mingle,
		add the sunflower oil and mingle,

		serve forth;
